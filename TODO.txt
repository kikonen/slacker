version: "2"
services:
  node:
    image: "node:14"
    user: "node"
    working_dir: /app
    environment:
    - NODE_ENV=production
    volumes:
    - ./:/app
    expose:
    - "8081"
    command: "npm start"


  nginx-proxy:
    image: jwilder/nginx-proxy:0.8.0
    volumes:
      - /var/run/docker.sock:/tmp/docker.sock:ro
    networks:
      - default
    ports:
      - "8080:80"
    volumes:
      - ./nginx-proxy/config:/etc/nginx
      - ./nginx-proxy/certs:/etc/ssl/private


server {
    listen 8080;
    server_name production_slacker;

    location /ui {
        rewrite /ui(.*) /origin-d$1;
        proxy_pass http://localhost:3000;
        proxy_redirect / /ui;
        sub_filter 'href="/' 'href="/ui';
    }
}
